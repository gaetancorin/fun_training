-------------- AIDE DOCKER COMPOSE ---------------
construire le docker compose:
docker-compose -p airflow up -d
docker-compose -p airflow up --build -d

redemarrer le docker compose:
docker-compose -p airflow start

arreter le docker compose:
docker-compose -p airflow stop

supprimer le docker compose et docker associés:
docker-compose -p airflow down
docker-compose -p airflow down -v

recréer le docker compose:
docker-compose -p airflow down
docker-compose -p airflow up -d

TUTORIEL PREMIER FONCTIONNEMENT

telecharger la dernière version stable de airflow:
https://airflow.apache.org/docs/apache-airflow/stable/docker-compose.yaml

retirer les dags par defaut:
sur le docker-compose
AIRFLOW__CORE__LOAD_EXAMPLES: 'false'

creer fichier .env et ecrire:
AIRFLOW_UID=50000

créer les containers et aller sur
http://localhost:8080/

se connecter avec
airflow
